# C99 Core Port — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port the S3M playback path to a C99 `core/` library and prove bit-identical output against the reference render via CTest sha256 comparison.

**Architecture:** Mechanical translation of `_work/dat_s3m.cpp`, `_work/efc_s3m.cpp`, and `_work/mixer.cpp` to C99 files under `core/`. Global parallel arrays are preserved. A thin `fx_load` / `fx_render_frames` / `fx_close` API in `engine/fx.c` wires everything together. A `tests/render_s3m/` binary renders `TEST.S3M` and CTest sha256-compares the output to `tests/reference_renders/TEST.wav`.

**Tech Stack:** C99, CMake 3.16+, CTest, sha256sum (system tool)

## Global Constraints

- C99 only in `core/` — no C++ syntax, no `//`-only comments, no `<dos.h>`, `<conio.h>`, `<stdlib.h>`, `<stdio.h>`.
- No `malloc`/`free` in core. Caller passes workspace buffer sized by `fx_workspace_size`.
- Fixed-width types everywhere: `uint8_t`, `uint16_t`, `uint32_t`, `int32_t`, `int64_t`, `size_t` from `<stdint.h>` / `<stddef.h>`.
- File-scope globals preserved exactly as in the original (no struct encapsulation yet).
- No `printf`, no `exit()` — errors return `fx_err` codes.
- Mix scratch buffer is `static int32_t mix_scratch[FX_INTERNAL_BLOCK * 2]` (internal to `mixer_scalar.c`). `FX_INTERNAL_BLOCK = 4096`.
- Translation rule table (apply to ALL files):

| Original | C99 |
|---|---|
| `unsigned long` | `uint32_t` |
| `signed long` | `int32_t` |
| `unsigned short` | `uint16_t` |
| `unsigned char` | `uint8_t` |
| `char *` buffer ptr | `uint8_t *` |
| `__int64` | `int64_t` |
| `#include <dos.h>` etc. | Delete |
| `extern "C" {}` | Delete |
| `#pragma aux ...` | Replace with C (see Task 2) |
| `printf(...)` | Delete |
| `exit(...)` | `return <error_code>` |
| `malloc`/`free` | Workspace carving (see Task 3) |
| `FILE *` / `fopen` | Buffer pointer + offset (see Task 3) |

- Reference render: `tests/reference_renders/TEST.wav` sha256 = `5edd49d54cd161dfac8c88ffbec30bf6fcadea4a2e3cd053db3dbb1c68c30834`
- Render config for CTest: 48000 Hz, stereo, S16, `interpolate=1`, `soft_clip=1`

The original `divide_64bit` uses a Watcom `#pragma aux` inline that does a 64/32-bit `div` instruction. The C99 equivalent:
- divisor = `(div_pos << 16) | (div_frac & 0xFFFF)`
- dividend = `((uint64_t)pos << 16) | (frac & 0xFFFF)`
- `g_division = (uint32_t)(dividend / divisor)`
- `g_div_fraction = (uint32_t)(dividend % divisor)`

```c
#include "calc.h"

volatile uint32_t g_division     = 0;
volatile uint32_t g_div_fraction = 0;

uint32_t s3m_calc_speed(uint32_t tempo, uint32_t mix_speed)
{
    uint32_t t = ((mix_speed * 10u) << 2) / (tempo * 4u);
    return (t + 2u) >> 2;
}

/*
 * 64/32-bit fixed-point division matching the original Watcom #pragma aux.
 * Replicates: shl ebx,16 / mov bx,cx (build 32-bit divisor)
 *             shld edx,eax,16 / shl eax,16 / mov ax,cx (build 64-bit dividend)
 *             div ebx
 */
void s3m_divide_64bit(uint32_t pos, uint32_t frac,
                      uint32_t div_pos, uint32_t div_frac)
{
    uint32_t divisor  = (div_pos << 16) | (div_frac & 0xFFFFu);
    uint64_t dividend = ((uint64_t)pos << 16) | (frac & 0xFFFFu);
    if (divisor == 0) {
        g_division     = 0xFFFFFFFFu;
        g_div_fraction = 0;
        return;
    }
    g_division     = (uint32_t)(dividend / divisor);
    g_div_fraction = (uint32_t)(dividend % divisor);
}
```

- Produces (called by engine/fx.c):
  - `s3m_workspace_bytes(const uint8_t *data, size_t size) → size_t`

**Workspace layout** (carved in order inside `s3m_load`):

```
ws_ptr offset 0:   uint8_t module_copy[size]              (writable copy for U8→S8)
                   + align to sizeof(uint8_t*)
ws_ptr next:       uint8_t *InstrumentPointer[InsNum]
ws_ptr next:       uint8_t *SamplePointer[InsNum]
ws_ptr next:       uint8_t *PatternPointer[PatNum]
ws_ptr next:       int32_t  MasterVolumeTable[16384 + 2]  (65536 + 8 bytes, centred)
```

`s3m_workspace_bytes` computes: `size + align(8) + InsNum*sizeof(uint8_t*)*2 + PatNum*sizeof(uint8_t*) + 65544 + 128`

- [ ] **Step 3: Continue `s3m.c` — workspace sizing and load function**

```c
/* ---- conversion helpers (replaces convert.cpp) ---- */
static void conv_u8m_to_s8m(uint8_t *data, uint32_t length) {
    uint32_t i;
    for (i = 0; i < length; i++) data[i] ^= 0x80u;
}
static void conv_u16m_to_s16m(uint8_t *data, uint32_t length) {
    uint16_t *s = (uint16_t *)data;
    uint32_t i;
    for (i = 0; i < length; i++) s[i] ^= 0x8000u;
}

/* ---- workspace sizing ---- */
size_t s3m_workspace_bytes(const uint8_t *data, size_t size)
{
    const s3m_header_t *hdr;
    size_t ptr_align, ws;
    if (size < 64) return 0;
    hdr = (const s3m_header_t *)data;
    ptr_align = sizeof(uint8_t *) - 1;
    ws  = size;
    ws  = (ws + ptr_align) & ~ptr_align;          /* align module copy */
    ws += (size_t)hdr->ins_num * sizeof(uint8_t *) * 2u;  /* ins + smp ptrs */
    ws += (size_t)hdr->pat_num * sizeof(uint8_t *);        /* pat ptrs */
    ws += 65544u;                                  /* MasterVolumeTable */
    ws += 128u;                                    /* alignment overhead */
    return ws;
}

/* ---- loader ---- */
int s3m_load(const uint8_t *data, size_t size,
             uint8_t *ws, size_t ws_size)
{
    const s3m_header_t *hdr = (const s3m_header_t *)data;
    uint8_t *p;
    size_t ptr_align, module_aligned;
    uint32_t channel, i;
    const uint16_t *tp;

    if (size < 64)                                   return -1;
    if (memcmp(hdr->scrm, "SCRM", 4))                return -1;
    if (hdr->value_1ah != 0x1a)                      return -1;
    if (hdr->type      != 16)                        return -1;
    if (hdr->ord_num < 2)                            return -1;
    if (hdr->ins_num  == 0)                          return -1;
    if (hdr->pat_num  == 0)                          return -1;
    if (ws_size < s3m_workspace_bytes(data, size))   return -2;

    /* ---- carve workspace ---- */
    ptr_align     = sizeof(uint8_t *) - 1;
    module_aligned = (size + ptr_align) & ~ptr_align;

    s_buf = ws;
    memcpy(s_buf, data, size);

    p     = ws + module_aligned;
    s_ins = (uint8_t **)p;
    p    += hdr->ins_num * sizeof(uint8_t *);
    s_smp = (uint8_t **)p;
    p    += hdr->ins_num * sizeof(uint8_t *);
    s_pat = (uint8_t **)p;
    p    += hdr->pat_num * sizeof(uint8_t *);
    /* align p to 8 before int32_t table */
    p     = (uint8_t *)(((uintptr_t)p + 7u) & ~(uintptr_t)7u);
    /* MasterVolumeTable: 16384 entries each side of centre + guard */
    g_master_vol_table = (int32_t *)p + 8192;  /* centre at index 0 */

    s_ins_num = hdr->ins_num;
    s_pat_num = hdr->pat_num;

    /* set pointers into module copy */
    s_chansets = s_buf + 0x40;
    s_order    = s_buf + 0x60;

    tp = (const uint16_t *)(s_buf + 0x60 + hdr->ord_num);
    for (i = 0; i < hdr->ins_num; i++) {
        s_ins[i] = s_buf + ((uint32_t)tp[i] << 4);
        {
            uint32_t para = (uint32_t)(*(uint16_t *)(s_ins[i] + 0x0e))
                          | ((uint32_t)(*(uint8_t  *)(s_ins[i] + 0x0d)) << 16);
            s_smp[i] = s_buf + (para << 4);
            *(uint8_t *)(s_ins[i] + 0x0d) = 0;  /* terminate sample name */
        }
    }

    tp = (const uint16_t *)(s_buf + 0x60 + hdr->ord_num + hdr->ins_num * 2u);
    for (i = 0; i < hdr->pat_num; i++) {
        if (tp[i] != 0)
            s_pat[i] = s_buf + ((uint32_t)tp[i] << 4);
        else
            s_pat[i] = NULL;
    }

    /* convert unsigned samples to signed */
    if (hdr->file_format_info == 2) {
        for (i = 0; i < hdr->ins_num; i++) {
            if (*(uint8_t *)s_ins[i] != 1) continue;
            uint8_t flags = *(uint8_t *)(s_ins[i] + 0x1f);
            uint32_t len  = *(uint32_t *)(s_ins[i] + 0x10);
            if ((flags & 4) == 0) {
                if ((flags & 2) == 0) conv_u8m_to_s8m(s_smp[i], len);
            } else {
                if ((flags & 2) == 0) conv_u16m_to_s16m(s_smp[i], len);
            }
        }
    }

    /* global state init */
    S3M_OrderNum       = hdr->ord_num;
    S3M_InstrumentNum  = hdr->ins_num;
    S3M_PatternNum     = hdr->pat_num;

    S3M_initvariables();
    S3M_initEffects();

    S3M_GlobalVolume        = hdr->global_volume;
    g_MasterVolume          = (uint32_t)(hdr->master_volume & 127u) * 256u;
    S3M_flag_stereo         = hdr->master_volume >> 7;
    S3M_flag_amigalimits    = (hdr->flags & 16u) >> 4;
    S3M_flag_st3volslides   = (hdr->flags & 64u) >> 6;
    if (hdr->tracker_version == 0x1300) S3M_flag_st3volslides = 1;

    S3M_tempo  = hdr->init_tempo;
    S3M_speed  = hdr->init_speed;

    S3M_channels = 0;
    for (channel = 0; channel < S3M_MAXCHANNELS; channel++)
        if ((s_chansets[channel] & 127u) < 16u) S3M_channels++;

    /* default panning */
    if (hdr->default_panning == 0xFC) {
        const uint8_t *defpan = s_order + hdr->ord_num
                              + hdr->ins_num * 2u + hdr->pat_num * 2u;
        for (channel = 0; channel < S3M_MAXCHANNELS; channel++) {
            if (S3M_flag_stereo)
                S3M_Panning[channel] = 0x03u + ((s_chansets[channel] & 15u) >> 3) * 0x09u;
            if (defpan[channel] & 32u)
                S3M_Panning[channel] = defpan[channel] & 15u;
        }
    }

    /* init timing */
    S3M_TickLength = s3m_calc_speed(S3M_tempo, g_MixSpeed);

    S3M_row = 0; S3M_jump = 0; S3M_tick = 0;
    S3M_tick_rest = S3M_TickLength;
    S3M_Order   = 0;
    S3M_Pattern = s_order[0];

    while (S3M_Pattern > 199) {
        S3M_Order++;
        S3M_Pattern = s_order[S3M_Order];
        if (S3M_Pattern == 255) return -1;
    }

    S3M_unpack_row(S3M_Pattern, S3M_row);
    S3M_read_row();

    mixer_calc_master_vol32(g_MasterVolume, g_master_vol_table);

    return 0;
}

{
    s_buf = NULL; s_ins = NULL; s_smp = NULL; s_pat = NULL;
    s_order = NULL; s_chansets = NULL;
    g_master_vol_table = NULL;
}

{
    uint32_t channel;
    g_ChannelLast = S3M_channels;

        g_ChannelActiv[channel]  = S3M_ChannelActiv[channel];

            g_ChannelSampleFraction[channel]  = S3M_SampleFraction[channel];

        if (S3M_ChannelActiv[channel]) {
            S3M_SamplePosition[channel] = g_ChannelSamplePosition[channel];

/*
 * Render one block of `block_frames` output frames.
 * This is the C99 equivalent of interrupt_S3M().
 * Caller (engine/fx.c) then calls mixer_convert_to_s16() on the scratch.
 */
void s3m_render_block(uint32_t block_frames)
{
    uint32_t toplay;
    S3M_buffer_rest = block_frames;
    g_MixerAddress  = 0;
    mixer_clear();

    while (S3M_buffer_rest > 0) {
        if (S3M_tick_rest <= S3M_buffer_rest) {
            toplay = S3M_tick_rest;
            s3m_to_mixer();
            mixer_do_pre_mixing(toplay);
            mixer_to_s3m();
            g_MixerAddress += toplay;
            S3M_tick++;
            if (S3M_tick == S3M_speed) {
                S3M_tick = 0;
                S3M_goRowOrder();
                S3M_unpack_row(S3M_Pattern, S3M_row);
                if (s_dat_ready == 1) S3M_read_row();
            } else {
                S3M_TickEffect();
            }
            S3M_buffer_rest -= toplay;
            S3M_tick_rest    = S3M_TickLength;
        } else {
            toplay = S3M_buffer_rest;
            s3m_to_mixer();
            mixer_do_pre_mixing(toplay);
            mixer_to_s3m();
            g_MixerAddress += toplay;
            S3M_tick_rest  -= toplay;
            S3M_buffer_rest = 0;
        }
    }
}

        vol_r = (g_ChannelVolumeRight[ch] * g_GlobalVolume + 128u) >> 8;
    } else {
        vol_l = vol_r = vol;
    }

    mb = s_mix_scratch + (g_flag_stereo ? mix_pos * 2 : mix_pos);

    if (g_ChannelSampleBits[ch] == 8) {
        s8 = (const uint8_t *)(uintptr_t)g_ChannelSampleAddress[ch];
        if (!g_flag_stereo) {
            for (i = 0; i < count; i++) {
                mb[i] += (int32_t)(int8_t)s8[pos] * (int32_t)vol;
                frac += (uint32_t)dfrac;
                pos  += (uint32_t)dpos + (frac >> 16);
                frac &= 0xFFFFu;
            }
        } else {
            for (i = 0; i < count; i++) {
                int32_t s = (int32_t)(int8_t)s8[pos];
                mb[i*2+0] += s * (int32_t)vol_l;
                mb[i*2+1] += s * (int32_t)vol_r;
                frac += (uint32_t)dfrac;
                pos  += (uint32_t)dpos + (frac >> 16);
                frac &= 0xFFFFu;
            }
        }
    } else {
        s16 = (const int16_t *)(uintptr_t)g_ChannelSampleAddress[ch];
        if (!g_flag_stereo) {
            for (i = 0; i < count; i++) {
                mb[i] += (int64_t)s16[pos] * vol >> 8;
                frac += (uint32_t)dfrac;
                pos  += (uint32_t)dpos + (frac >> 16);
                frac &= 0xFFFFu;
            }
        } else {
            for (i = 0; i < count; i++) {
                int32_t s = s16[pos];
                mb[i*2+0] += (int64_t)s * vol_l >> 8;
                mb[i*2+1] += (int64_t)s * vol_r >> 8;
                frac += (uint32_t)dfrac;
                pos  += (uint32_t)dpos + (frac >> 16);
                frac &= 0xFFFFu;
            }
        }
    }
}
```

**IMPORTANT — volume scaling:** The 8-bit and 16-bit paths must produce identical accumulation scaling to what the original ASM does, or the output will not match. Verify against the reference render. If there is a mismatch, adjust the `>> 8` shift or the volume multiplier to match the ASM behaviour decoded from `_work/mixr_32n.asm` and `_work/mixr_32i.asm`.

For the original 32-bit mixer (`mix8Mto32M`): the ASM reads `(int8_t)sample * volume` via a 256×256 lookup table (`Voltable8`) computed as `Sample * Volume * 256`. So the raw accumulation is `(int8_t)sample * volume * 256`. The final `DoMasterVolumeCalculations` path divides by `0x800000` (= 8388608). So: `mix_buf[i] += (int8_t)sample * vol * 256` for 8-bit.

For `mix16Mto32M`: the ASM does `imul eax, Volume` where `eax = (int16_t)sample`. No table, direct multiply. So: `mix_buf[i] += (int16_t)sample * vol`. No `>> 8`.

Correct the inner loops accordingly:
```c
/* 8-bit: */ mb[i] += (int32_t)(int8_t)s8[pos] * (int32_t)vol * 256;
/* 16-bit: */ mb[i] += (int32_t)s16[pos] * (int32_t)vol;
```

```c
void mixer_convert_to_s16(int16_t *out, uint32_t frame_count,
                          const int32_t *master_vol_table)
{
    /*
     * Translates MasterVol16for32ss:
     *   for each 32-bit mix sample v:
     *     idx    = v >> 18  (signed shift, range -8192..8191)
     *     scaled = (int64_t)v * master_vol_table[idx]
     *     result = scaled / 0x800000 + 0x8000
     *     clip to [0, 0xFFFF]
     *     out[i] = (int16_t)(result - 32768)
     */
    uint32_t n = frame_count * (g_flag_stereo ? 2u : 1u);
    uint32_t i;
    for (i = 0; i < n; i++) {
        int32_t v   = s_mix_scratch[i];
        int32_t idx = v >> 18;             /* signed arithmetic shift */
        int64_t sc  = (int64_t)v * master_vol_table[idx];
        int32_t r   = (int32_t)(sc / 0x800000L) + 0x8000;
        if (r & (int32_t)0xFFFF0000) {
            r = (r < 0) ? 0 : 0xFFFF;
        }
        out[i] = (int16_t)(r - 32768);
    }
}
```

**If `g_flag_soft_clip == 0` (hard clip):** the hard-clip path (`MasterVol16for32sh`) is simpler — `master_vol_table` is filled with a constant `g_MasterVolume` at all indices, so `scaled / 0x800000` is linear. A compile-time or runtime branch on `g_flag_soft_clip` selects between calling `mixer_calc_master_vol32` (which fills the table with a soft curve) vs. filling it with a constant.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "fx/fx.h"

/* Minimal WAV header writer */
static void write_wav_header(FILE *f, uint32_t sample_rate,
                             uint16_t channels, uint32_t n_frames)
{
    uint32_t data_bytes = n_frames * channels * 2u;
    uint32_t riff_size  = 36u + data_bytes;
    uint16_t bits = 16, fmt_pcm = 1;
    uint32_t byte_rate  = sample_rate * channels * 2u;
    uint16_t block_align= (uint16_t)(channels * 2u);

    fwrite("RIFF", 1, 4, f);
    fwrite(&riff_size,   4, 1, f);
    fwrite("WAVEfmt ", 1, 8, f);
    uint32_t fmt_size = 16;
    fwrite(&fmt_size,    4, 1, f);
    fwrite(&fmt_pcm,     2, 1, f);
    fwrite(&channels,    2, 1, f);
    fwrite(&sample_rate, 4, 1, f);
    fwrite(&byte_rate,   4, 1, f);
    fwrite(&block_align, 2, 1, f);
    fwrite(&bits,        2, 1, f);
    fwrite("data",       1, 4, f);
    fwrite(&data_bytes,  4, 1, f);
}

int main(int argc, char **argv)
{
    FILE *fin, *fout;
    long  file_size;
    uint8_t *module_data;
    uint8_t *workspace;
    size_t   ws_bytes;
    fx_err   err;

    static const fx_config cfg = {
        48000, 2, FX_OUTPUT_S16, 1, 1  /* 48kHz stereo S16 interp softclip */
    };

    #define CHUNK_FRAMES 1024
    static int16_t pcm_buf[CHUNK_FRAMES * 2];

    /* buffers for rendered output (dynamic realloc) */
    int16_t *all_pcm  = NULL;
    size_t   all_cap  = 0;
    size_t   all_used = 0;

    size_t rendered;
    uint32_t total_frames = 0;

    if (argc < 3) {
        fprintf(stderr, "Usage: render_s3m <input.s3m> <output.wav>\n");
        return 1;
    }

    /* read module */
    fin = fopen(argv[1], "rb");
    if (!fin) { perror(argv[1]); return 1; }
    fseek(fin, 0, SEEK_END);
    file_size = ftell(fin);
    fseek(fin, 0, SEEK_SET);
    module_data = (uint8_t *)malloc((size_t)file_size);
    if (!module_data) { fputs("OOM\n", stderr); return 1; }
    if (fread(module_data, 1, (size_t)file_size, fin) != (size_t)file_size) {
        fputs("read error\n", stderr); return 1;
    }
    fclose(fin);

    /* size and allocate workspace */
    err = fx_workspace_size(module_data, (size_t)file_size, &ws_bytes);
    if (err != FX_OK) { fprintf(stderr, "workspace_size error %d\n", err); return 1; }
    workspace = (uint8_t *)malloc(ws_bytes);
    if (!workspace) { fputs("OOM workspace\n", stderr); return 1; }

    /* load */
    err = fx_load(module_data, (size_t)file_size, workspace, ws_bytes);
    if (err != FX_OK) { fprintf(stderr, "fx_load error %d\n", err); return 1; }

    /* render loop */
    while ((rendered = fx_render_frames(pcm_buf, CHUNK_FRAMES, &cfg)) > 0) {
        size_t samples = rendered * cfg.channels;
        if (all_used + samples > all_cap) {
            all_cap = (all_cap == 0) ? (1 << 20) : all_cap * 2;
            all_pcm = (int16_t *)realloc(all_pcm, all_cap * sizeof(int16_t));
            if (!all_pcm) { fputs("OOM pcm\n", stderr); return 1; }
        }
        memcpy(all_pcm + all_used, pcm_buf, samples * sizeof(int16_t));
        all_used    += samples;
        total_frames += (uint32_t)rendered;
    }

    /* write WAV */
    fout = fopen(argv[2], "wb");
    if (!fout) { perror(argv[2]); return 1; }
    write_wav_header(fout, cfg.sample_rate, cfg.channels, total_frames);
    fwrite(all_pcm, sizeof(int16_t), all_used, fout);
    fclose(fout);

    free(all_pcm);
    fx_close();
    free(workspace);
    free(module_data);

    fprintf(stderr, "Rendered %u frames → %s\n", total_frames, argv[2]);
    return 0;
}
